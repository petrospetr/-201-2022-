def rk4Algor(t,h,N,y,f):
            k1=np.zeros(N)
            k2=np.zeros(N)
            k3=np.zeros(N)
            k4=np.zeros(N)
            k1 = h*f(t,y)
            k2 = h*f(t+h/2.,y+k1/2.)
            k3 = h*f(t+h/2.,y+k2/2.)
            k4 = h*f(t+h,y+k3)
            y=y+(k1+2*(k2+k3)+k4)/6.
            return y

fig1,ax1=plt.subplots(len(ns))
j=len(ns)-1
rVec=np.zeros((int(5/0.01)),float)
psiVec=np.zeros((int(5/0.01)),float)
for n in ns:
    y[0]=1
    y[1]=10**(-8)

    i=0
    f(0.0,y)
    dr=0.01
    
    for r in np.arange(0,5,dr):
        rVec[i]=r
        y=rk4Algor(r,dr,2,y,f)
        
        psiVec[i]=y[0]
        i=i+1
        
    ax1[j].plot(rVec,psiVec, label="n=%d"%n,color="red")  
    ax1[j].grid()

    ax1[j].legend(loc="lower right")
    j-=1
